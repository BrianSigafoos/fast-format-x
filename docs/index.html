.fast-format-x.yaml

# Required config file in your project root
version: 1

tools:
  # JavaScript / TypeScript / React
  - name: standard
    include: ["**/*.js", "**/*.jsx", "**/*.ts", "**/*.tsx"]
    cmd: npx
    args: [standard, --fix]

  # Markdown / JSON
  - name: prettier
    include: ["**/*.md", "**/*.json"]
    cmd: npx
    args: [prettier, --write]

  # Ruby
  - name: rubocop
    include: ["**/*.rb"]
    exclude: ["vendor/**"]
    cmd: bundle
    args: [exec, rubocop, -a]

  # Rust
  - name: rustfmt
    include: ["**/*.rs"]
    cmd: cargo
    args: [fmt, --]

  # Go
  - name: gofmt
    include: ["**/*.go"]
    cmd: gofmt
    args: [-w]
