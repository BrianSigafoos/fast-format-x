.fast-format-x.yaml

# Required config file in your project root
version: 1

tools:
  # Ruby
  - name: rubocop
    include: ["**/*.rb"]
    exclude: ["vendor/**"]
    cmd: bundle
    args: [exec, rubocop, -a]

  # JavaScript
  - name: prettier
    include: ["**/*.js", "**/*.ts"]
    cmd: npx
    args: [prettier, --write]

  # Rust
  - name: rustfmt
    include: ["**/*.rs"]
    cmd: cargo
    args: [fmt, --]

  # Kotlin
  - name: ktlint
    include: ["**/*.kt", "**/*.kts"]
    cmd: ktlint
    args: [-F]

  # Go
  - name: gofmt
    include: ["**/*.go"]
    cmd: gofmt
    args: [-w]
