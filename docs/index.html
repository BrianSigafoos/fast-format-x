Configuration

version: 1

tools:
  # Ruby
  - name: rubocop
    include: ["**/*.rb"]
    exclude: ["vendor/**"]
    cmd: bundle
    args: [exec, rubocop, -a]

  # JavaScript
  - name: prettier
    include: ["**/*.js", "**/*.ts"]
    cmd: npx
    args: [prettier, --write]

  # Rust
  - name: rustfmt
    include: ["**/*.rs"]
    cmd: cargo
    args: [fmt, --]
