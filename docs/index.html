.fast-format-x.yaml

# Config file in your project root. Remove tools you don't need.
version: 1

tools:
  # Go
  - name: gofmt
    include: ["**/*.go"]
    cmd: gofmt
    args: [-w]

  # JavaScript / TypeScript / React
  - name: standard
    include: ["**/*.js", "**/*.jsx", "**/*.ts", "**/*.tsx"]
    cmd: npx
    args: [standard, --fix]

  # Kotlin
  - name: ktlint
    include: ["**/*.kt", "**/*.kts"]
    cmd: ktlint
    args: [-F]

  # Python (Ruff - linter + formatter)
  - name: ruff
    include: ["**/*.py"]
    cmd: uv
    args: [run, ruff, check, --fix, .]

  # Ruby
  - name: rubocop
    include: ["**/*.rb"]
    cmd: bundle
    args: [exec, rubocop, -A]

  # Rust
  - name: rustfmt
    include: ["**/*.rs"]
    cmd: cargo
    args: [fmt, --]

  # Swift
  - name: swiftformat
    include: ["**/*.swift"]
    cmd: swiftformat
    args: [.]

  # Web: CSS, Markdown, JSON, YAML, HTML
  - name: prettier
    include: ["**/*.css", "**/*.md", "**/*.json", "**/*.yaml", "**/*.html"]
    cmd: npx
    args: [prettier, --write]
